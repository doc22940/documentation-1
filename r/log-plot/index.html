    Log Axes

    library(plotly)
d <- diamonds[sample(nrow(diamonds), 1000), ]
# without log scales
p <- plot_ly(d, x = ~carat, y = ~price) %>% add_markers()

p

    # with log scales
p <- layout(p, xaxis = list(type = "log"),
       yaxis = list(type = "log"))

p
